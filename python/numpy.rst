Numpy Slicing
========================================

.. code-block:: python

    >>> import numpy as np
    >>> data = np.mat('1 2 3; 4 5 6; 7 8 9')
    >>> data
    matrix([[1, 2, 3],
            [4, 5, 6],
            [7, 8, 9]])
    >>> data[:, 0]   # row : all, col : 0
    matrix([[1],
            [4],
            [7]])
    >>> data[:, (0, 2)]         # row : all, col : (0, 2)
    matrix([[1, 3],
            [4, 6],
            [7, 9]])
    >>> data[(0, 2), (1, 1)]    # parameters : rows & cols, access (0, 1) & (2, 1)
    matrix([[2, 8]])
    >>> data[(0, 2),
    ...      (1, 1)]
    matrix([[2, 8]])
    >>> data[((0, 0), (1, 2)),
    ...      ((2, 1), (2, 1))]
    matrix([[3, 2],
            [6, 8]])
    >>> data[(((0, 0), (1, 2)), ((0, 0), (1, 2))),
    ...      (((2, 1), (2, 1)), ((2, 1), (2, 1)))]
    matrix([[[3, 2],
            [6, 8]],

            [[3, 2],
            [6, 8]]])
    >>> data[ np.mat('0 0; 1 2') ]
    matrix([[[1, 2, 3],
            [1, 2, 3]],

            [[4, 5, 6],
            [7, 8, 9]]])
